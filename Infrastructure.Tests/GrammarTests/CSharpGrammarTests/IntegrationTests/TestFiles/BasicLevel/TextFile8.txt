using System;

namespace MyNamespace
{
    class Program
    {
        static void Main()
        {
            DerivedClassA instanceA = new DerivedClassA();
            DerivedClassB instanceB = new DerivedClassB();

            instanceA.MethodA();
            instanceB.MethodB();
        }
    }

    class BaseClass
    {
        protected void CommonMethod()
        {
        }
    }

    class DerivedClassA : BaseClass
    {
        public void MethodA()
        {
            base.CommonMethod();
            AnotherMethodA();
        }

        private void AnotherMethodA()
        {
        }
    }

    class DerivedClassB : BaseClass
    {
        public void MethodB()
        {
            this.CommonMethod();
            AnotherMethodB();
        }

        private void AnotherMethodB()
        {
        }
    }
}
